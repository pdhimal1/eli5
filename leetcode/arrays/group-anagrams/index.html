The trick: sort each word! "eat" â†’ "aet", "tea" â†’ "aet", "tan" â†’ "ant". All words with the same sorted version are anagrams! Use sorted word as the hash key.

ðŸ’» The Clean Solution

def group_anagrams(strs):
    # A box with labeled compartments
    groups = {}
    
    # Process each word
    for word in strs:
        # Sort letters to create a "key"
        # "eat" -> "aet", "tea" -> "aet"
        key = ''.join(sorted(word))
        
        # Put word in the right compartment
        if key not in groups:
            groups[key] = []
        groups[key].append(word)
    
    # Return all the groups
    return list(groups.values())

# Try it!
words = ["eat", "tea", "tan", "ate", "nat", "bat"]
print(group_anagrams(words))
# Output: [["eat","tea","ate"], ["tan","nat"], ["bat"]]
